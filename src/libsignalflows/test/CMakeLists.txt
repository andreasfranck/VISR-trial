SET( APPLICATION_NAME libsignalflow_test )

# Make the current source directory known to the source code to create absolute file paths.
ADD_DEFINITIONS( -DCMAKE_CURRENT_SOURCE_DIR="${CMAKE_CURRENT_SOURCE_DIR}" )
ADD_DEFINITIONS( -DCMAKE_SOURCE_DIR="${CMAKE_SOURCE_DIR}" )

ADD_EXECUTABLE( ${APPLICATION_NAME} 
baseline_renderer.cpp
test_main.cpp )

TARGET_LINK_LIBRARIES(${APPLICATION_NAME} signalflows)
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} rcl)
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} objectmodel)
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} panning)
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} pml )
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} rbbl)
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} visr_impl)
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} ril)
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} rrl)
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} efl)
TARGET_LINK_LIBRARIES( ${APPLICATION_NAME} ${Boost_UNIT_TEST_FRAMEWORK_LIBRARY} )
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} ${Boost_FILESYSTEM_LIBRARY} )
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} ${Boost_THREAD_LIBRARY} )
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} ${Boost_PROGRAM_OPTIONS_LIBRARY} )
TARGET_LINK_LIBRARIES(${APPLICATION_NAME} ${Boost_SYSTEM_LIBRARY} )

TARGET_LINK_LIBRARIES(${APPLICATION_NAME} ${CMAKE_THREAD_LIBS_INIT} )

set_target_properties( ${APPLICATION_NAME} PROPERTIES 
  RUNTIME_OUTPUT_DIRECTORY  ${CMAKE_CURRENT_BINARY_DIR}/test_binaries)

add_test(NAME ${APPLICATION_NAME} WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} 
         COMMAND ${CMAKE_CURRENT_BINARY_DIR}/test_binaries/${APPLICATION_NAME} )
